PRINT
print([expression]);
ex.
print("hello world"); -> hello world
print(1 + 1); -> 2
print(1 + 1 == 2); -> True
i32 a = 5;
print(a); -> 5

VARIABLES
[variable type] [variable name] = [value];
ex.
String s = "hello world";
i32 eleven = 11;
Bool red = false;

WHILE
while([expression -> condition]){
    [lines to be executed]
}
ex.
while(true){
    print("foo");
}
Bool a = true;
while(a){
    print("bar");
}

IF
if([expression -> condition]){
    [lines to be executed]
}
ex.
if(true){
    print("foo");
}
Bool a = true;
if(a){
    print("bar");
}
